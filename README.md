# nuxeo-sandbox

# > parent + core :
nuxeo bootstrap multi-module
#id : com.mlefree.nuxeo.sandbox
#nuxeo version : 11.1-SNAPSHOT or 11.5.97 or ... ?

# > package :
nuxeo bootstrap package

# > link nuxeo studio mleprevost-SANDBOX :
